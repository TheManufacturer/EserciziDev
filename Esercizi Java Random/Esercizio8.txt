public class Main {
    public static void main(String[] args) {
        int limite = 5;
        int somma=0;
        System.out.println(stampaSomma(limite));
    }

    public static int stampaSomma(int lim) {
        int max=1;
        int som=0;
        while(max<lim){
            som=som+max;
            max++;
        }
        return som;
    }
}
